Practical-30: Multiple Classes in a Package with Parameter Using Scanner method

p-30 folder > abc folder > xyz folder > xyz.java


package abc.xyz;

public class xyz
{
    public void sum(int a, int b)
    {
        System.out.println("Sum Is: " + (a + b));
    }
}

p-30 folder > abc folder > abc.java


package abc;

import abc.xyz.xyz;

public class abc extends xyz
{
    public void mul(int a, int b)
    {
        System.out.println("Multiplication Is: " + (a * b));
    }
}

p-30 folder > demo.java


import abc.abc;
import java.util.Scanner;

class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter Value-1: ");
        int a = sc.nextInt();

        System.out.print("Enter Value-2: ");
        int b = sc.nextInt();

        aa.sum(a, b);
        aa.mul(a, b);
    }
}

Output:

Enter Value-1: 5
Enter Value-2: 4
Sum Is: 9
Multiplication Is: 20
