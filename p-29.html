Practical-29 : Multiple Class in Package with parameter

p-29 folder > abc folder > xyz folder > xyz.java


package abc.xyz;
public class xyz
{
    public void sum(int a,int b)
    {
        System.out.println("Ans="+(a+b));
    }
}

p-29 folder > abc folder > abc.java


package abc;
import abc.xyz.xyz;
public class abc extends xyz
{
    public void mul(int a,int b)
    {
        System.out.println("Ans="+(a*b));
    }
}

p-29 folder > demo.java


import abc.abc;
class demo
{   
    public static void main(String args[])
    {
        abc aa=new abc();
        aa.sum(3,2);
        aa.mul(5,4);    
    }
}

Output:

Ans=5 Ans=20
