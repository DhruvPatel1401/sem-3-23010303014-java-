Practical-28 : Multiple Class in Package

p-28 folder > abc folder > xyz folder > xyz.java


package abc.xyz;
public class xyz
{
    public void sum()
    {
        System.out.println("Java");
    }
}

p-28 folder > abc folder > abc.java


package abc;
import abc.xyz.xyz;
public class abc extends xyz
{
    public void mul()
    {
        System.out.println("Python");
    }
}

p-28 folder > demo.java


import abc.abc;
class demo
{   
    public static void main(String args[])
    {
        abc aa=new abc();
        aa.sum();
        aa.mul();   
    }
}

Output:

Java Python
